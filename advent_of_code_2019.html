def run_intcode(array):
    values = array[:]
    i = 0
    N = len(values)
    while i < N:
        if values[i] == 99:
            return values
        elif values[i] == 1:
            values[values[i+3]] = values[values[i+1]] + values[values[i+2]]
            i += 4
        elif values[i] == 2:
            values[values[i+3]] = values[values[i+1]] * values[values[i+2]]
            i += 4
        else:
            return "Error: {} is wrong opcode.".format(values[i])
    return "Error: program has no end opcode."

def calc_path(directions):
    path = [(0,0)]
    for step in directions:
        last_pos = path[-1]
        jump = int(step[1:])
        if step[0] == 'R':
            for i in range(1, jump+1):
                path.append((last_pos[0]+i, last_pos[1]))
        elif step[0] == 'L':
            for i in range(1, jump+1):
                path.append((last_pos[0]-i, last_pos[1]))
        elif step[0] == 'U':
            for i in range(1, jump+1):
                path.append((last_pos[0], last_pos[1]+i))
        elif step[0] == 'D':
            for i in range(1, jump+1):
                path.append((last_pos[0], last_pos[1]-i))
    return path


def get_bbox(path):
    minx = min(pos[0] for pos in path)
    maxx = max(pos[0] for pos in path)
    miny = min(pos[1] for pos in path)
    maxy = max(pos[1] for pos in path)
    
    return (minx, maxx, miny, maxy)


def filter_path(path, minx=None, maxx=None, miny=None, maxy=None):
    p = path[:]
    if minx:
        p = [pos for pos in p if pos[0]>=minx]
    if maxx:
        p = [pos for pos in p if pos[0]<=maxx]
    if miny:
        p = [pos for pos in p if pos[1]>=miny]
    if maxy:
        p = [pos for pos in p if pos[1]<=maxy]
    return p

        
def get_crossings(path1, path2):
    bbox1 = get_bbox(path1)
    bbox2 = get_bbox(path2)
    # print(bbox1)
    # print(bbox2)
    
    minx = max(bbox1[0], bbox2[0])
    maxx = min(bbox1[1], bbox2[1])
    miny = max(bbox1[2], bbox2[2])
    maxy = min(bbox1[3], bbox2[3])
    # print(minx, maxx, miny, maxy)

    # p1 = filter_path(path1, minx, maxx, miny, maxy)
    # p2 = filter_path(path2, minx, maxx, miny, maxy)
    # crossings = [pos for pos in p1 if pos in p2]

    p1 = set(filter_path(path1, minx, maxx, miny, maxy))
    p2 = set(filter_path(path2, minx, maxx, miny, maxy))
    crossings = p1.intersection(p2)
    crossings.remove((0, 0))
    
    return list(crossings)


def get_cross_distance(directions1, directions2):
    path1 = calc_path(directions1)
    path2 = calc_path(directions2)
        
    crossings = get_crossings(path1, path2)
    # print(crossings)
    distances = [abs(pos[0])+abs(pos[1]) for pos in crossings]
    # print(distances)
    return min(distances)

def is_valid_password(num):
    if num < 100000 or num > 999999:
        return False

    str_num = str(num)
    list_num = [int(u) for u in str_num]
    
    for i in range(1, 6):
        if list_num[i] < list_num[i-1]:
            return False
    
    for i in range(1, 6):
        if list_num[i] == list_num[i-1]:
            return True
    
    return False

def is_valid_password(num):
    if num < 100000 or num > 999999:
        return False

    list_num = [int(u) for u in str(num)]
    
    for i in range(1, 6):
        if list_num[i] < list_num[i-1]:
            return False
        
    distances = []
    start = 0
    i = 1
    while i < 6:
        if list_num[i] != list_num[i-1]:
            distances.append(i-start)
            start = i
        i += 1
    distances.append(i-start)

    if 2 in distances:
        return True
    else:
        return False

def get_opcode(value):
    opcode = value % 100
    if opcode not in (1, 2, 3, 4, 99):
        raise ValueError("ERROR: {} is wrong opcode".format(opcode))
    
    list_modes = []
    modes = value // 100
    for i in range(3):
        temp_mode, modes = modes % 10, modes // 10
        list_modes.append(temp_mode)

    if any(True if u not in (0, 1) else False for u in list_modes):
        raise ValueError("ERROR: wrong parameter code in {}".format(value))
    
    return opcode, list_modes

def run_intcode(array, input_nums=None):
    values = array[:]
    if input_nums:
        inputs = input_nums[:]
    output = []

    i = 0
    N = len(values)
    while i < N:
        opcode, modes = get_opcode(values[i])
        # print(opcode, modes)

        if opcode == 99:
            # print(output)
            return output
        elif opcode == 1:
            if modes[2] == 0:
                val1 = get_param(values, i, 1, modes)
                val2 = get_param(values, i, 2, modes)
                values[values[i+3]] = val1 + val2
            i += 4
        elif opcode == 2:
            if modes[2] == 0:
                val1 = get_param(values, i, 1, modes)
                val2 = get_param(values, i, 2, modes)
                values[values[i+3]] = val1 * val2
            i += 4
        elif opcode == 3:
            if modes[0] == 0:
                values[values[i+1]] = inputs.pop(0)
            i += 2
        elif opcode == 4:
            val1 = get_param(values, i, 1, modes)
            output.append(val1)
            i += 2
        elif opcode == 5:
            val1 = get_param(values, i, 1, modes)
            val2 = get_param(values, i, 2, modes)
            if val1 != 0:
                i = val2
            else:
                i += 3
        elif opcode == 6:
            val1 = get_param(values, i, 1, modes)
            val2 = get_param(values, i, 2, modes)
            if val1 == 0:
                i = val2
            else:
                i += 3
        elif opcode == 7:
            val1 = get_param(values, i, 1, modes)
            val2 = get_param(values, i, 2, modes)
            if modes[2] == 0:
                if val1 < val2:
                    values[values[i+3]] = 1
                else:
                    values[values[i+3]] = 0
            i += 4
        elif opcode == 8:
            val1 = get_param(values, i, 1, modes)
            val2 = get_param(values, i, 2, modes)
            if modes[2] == 0:
                if val1 == val2:
                    values[values[i+3]] = 1
                else:
                    values[values[i+3]] = 0
            i += 4

    return output
